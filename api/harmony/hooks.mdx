# Hooks

Harmony provides custom React hooks for common UI patterns and state management.

## useIsMobile

A hook that detects whether the viewport is below the mobile breakpoint (768px).

```tsx
import { useIsMobile } from '@gigadrive/harmony';

function MyComponent() {
  const isMobile = useIsMobile();

  return (
    <div>
      {isMobile ? (
        <MobileNavigation />
      ) : (
        <DesktopNavigation />
      )}
    </div>
  );
}
```

### Return Value

<ResponseField name="isMobile" type="boolean">
  `true` when viewport width is less than 768px, `false` otherwise. Initially `undefined` during SSR, then updates on mount.
</ResponseField>

### Implementation Details

- Uses `window.matchMedia` for efficient resize detection
- Updates only when crossing the 768px breakpoint (not on every resize)
- SSR-safe: returns `false` on initial render, updates in `useEffect`
- Automatically cleans up event listeners on unmount

### Example: Responsive Component

```tsx
import { useIsMobile } from '@gigadrive/harmony';
import { Dialog, Drawer } from '@gigadrive/harmony';

function ResponsiveModal({ children }) {
  const isMobile = useIsMobile();
  
  // Use Drawer on mobile, Dialog on desktop
  const Component = isMobile ? Drawer : Dialog;
  
  return (
    <Component>
      {children}
    </Component>
  );
}
```

### Breakpoint Constant

The mobile breakpoint is defined as:

```tsx
const MOBILE_BREAKPOINT = 768; // pixels
```

This matches Tailwind CSS's default `md` breakpoint.

## useDataTable

A hook for managing server-side data table state including pagination, sorting, and filtering.

```tsx
import { useDataTable } from '@gigadrive/harmony';

function UsersTable() {
  const table = useDataTable({
    fetchData: async ({ pagination, sorting, filters }) => {
      const response = await fetch('/api/users', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ pagination, sorting, filters }),
      });
      return response.json();
    },
    initialPagination: { pageIndex: 0, pageSize: 10 },
  });

  return (
    <DataTable
      columns={columns}
      data={table.data}
      pagination={table.pagination}
      onPaginationChange={table.setPagination}
      sorting={table.sorting}
      onSortingChange={table.setSorting}
      totalRows={table.totalRows}
      isLoading={table.isLoading}
    />
  );
}
```

### Options

<ParamField path="fetchData" type="function" required>
  Async function that fetches data from the server. Receives an object with `pagination`, `sorting`, and `filters`. Must return `{ data: TData[], totalRows: number }`.
</ParamField>

<ParamField path="initialPagination" type="PaginationState" default="{ pageIndex: 0, pageSize: 10 }">
  Initial pagination state with `pageIndex` (0-based) and `pageSize`.
</ParamField>

<ParamField path="initialSorting" type="SortingState | null" default={null}>
  Initial sorting state with `id` (column ID) and `direction` (`'asc'` or `'desc'`).
</ParamField>

<ParamField path="initialFilters" type="TFilters" default="{}">
  Initial filter values (generic type - shape defined by your application).
</ParamField>

<ParamField path="fetchOnMount" type="boolean" default={true}>
  Whether to automatically fetch data when component mounts.
</ParamField>

### Return Value

<ResponseField name="data" type="TData[]">
  Current page of data returned from the server.
</ResponseField>

<ResponseField name="totalRows" type="number">
  Total number of rows across all pages (for pagination calculation).
</ResponseField>

<ResponseField name="isLoading" type="boolean">
  `true` when data is being fetched.
</ResponseField>

<ResponseField name="error" type="Error | null">
  Error object if fetch failed, `null` otherwise.
</ResponseField>

<ResponseField name="pagination" type="PaginationState">
  Current pagination state `{ pageIndex: number, pageSize: number }`.
</ResponseField>

<ResponseField name="setPagination" type="(pagination: PaginationState) => void">
  Update pagination state (triggers refetch).
</ResponseField>

<ResponseField name="sorting" type="SortingState | null">
  Current sorting state `{ id: string, direction: 'asc' | 'desc' }` or `null`.
</ResponseField>

<ResponseField name="setSorting" type="(sorting: SortingState | null) => void">
  Update sorting state (triggers refetch).
</ResponseField>

<ResponseField name="filters" type="TFilters">
  Current filter values.
</ResponseField>

<ResponseField name="setFilters" type="(filters: TFilters | ((prev: TFilters) => TFilters)) => void">
  Update filter values (triggers refetch). Supports functional updates.
</ResponseField>

<ResponseField name="refetch" type="() => Promise<void>">
  Manually trigger a data refetch.
</ResponseField>

### Type Definitions

```typescript
interface PaginationState {
  pageIndex: number; // 0-based page index
  pageSize: number;  // Number of items per page
}

interface SortingState {
  id: string;                    // Column ID
  direction: 'asc' | 'desc';     // Sort direction
}

interface UseDataTableOptions<TData, TFilters> {
  fetchData: (params: {
    pagination: PaginationState;
    sorting: SortingState | null;
    filters: TFilters;
  }) => Promise<{
    data: TData[];
    totalRows: number;
  }>;
  initialPagination?: PaginationState;
  initialSorting?: SortingState | null;
  initialFilters?: TFilters;
  fetchOnMount?: boolean;
}
```

### Example: With Filters

```tsx
interface UserFilters {
  search: string;
  role: string | null;
  status: 'active' | 'inactive' | null;
}

function UsersTable() {
  const table = useDataTable<User, UserFilters>({
    fetchData: async ({ pagination, sorting, filters }) => {
      const params = new URLSearchParams({
        page: String(pagination.pageIndex),
        pageSize: String(pagination.pageSize),
        search: filters.search,
        ...(filters.role && { role: filters.role }),
        ...(filters.status && { status: filters.status }),
        ...(sorting && {
          sortBy: sorting.id,
          sortDir: sorting.direction,
        }),
      });
      
      const res = await fetch(`/api/users?${params}`);
      return res.json();
    },
    initialFilters: {
      search: '',
      role: null,
      status: null,
    },
  });

  return (
    <div>
      <div className="mb-4 flex gap-2">
        <Input
          placeholder="Search users..."
          value={table.filters.search}
          onChange={(e) => table.setFilters(prev => ({
            ...prev,
            search: e.target.value,
          }))}
        />
        <Select
          value={table.filters.role ?? ''}
          onValueChange={(role) => table.setFilters(prev => ({
            ...prev,
            role: role || null,
          }))}
        >
          <SelectTrigger className="w-[180px]">
            <SelectValue placeholder="Filter by role" />
          </SelectTrigger>
          <SelectContent>
            <SelectItem value="">All roles</SelectItem>
            <SelectItem value="admin">Admin</SelectItem>
            <SelectItem value="user">User</SelectItem>
          </SelectContent>
        </Select>
      </div>
      
      <DataTable
        columns={columns}
        data={table.data}
        pagination={table.pagination}
        onPaginationChange={table.setPagination}
        sorting={table.sorting}
        onSortingChange={table.setSorting}
        totalRows={table.totalRows}
        isLoading={table.isLoading}
      />
    </div>
  );
}
```

### Example: Manual Refetch

```tsx
function UsersTable() {
  const table = useDataTable({
    fetchData: fetchUsers,
    fetchOnMount: true,
  });

  const handleDeleteUser = async (userId: string) => {
    await fetch(`/api/users/${userId}`, { method: 'DELETE' });
    await table.refetch(); // Refresh the table
  };

  return (
    <div>
      <Button onClick={table.refetch}>Refresh</Button>
      <DataTable
        columns={columns}
        data={table.data}
        pagination={table.pagination}
        onPaginationChange={table.setPagination}
        totalRows={table.totalRows}
        isLoading={table.isLoading}
      />
    </div>
  );
}
```

### Example: Error Handling

```tsx
function UsersTable() {
  const table = useDataTable({
    fetchData: fetchUsers,
  });

  if (table.error) {
    return (
      <Alert variant="error">
        <AlertTitle>Failed to load users</AlertTitle>
        <AlertDescription>
          {table.error.message}
          <Button onClick={table.refetch}>Retry</Button>
        </AlertDescription>
      </Alert>
    );
  }

  return <DataTable {...table} columns={columns} />;
}
```

### Behavior Details

#### Automatic Refetching

The hook automatically refetches data when:
- Pagination changes (page index or page size)
- Sorting changes
- Filters change
- Component mounts (if `fetchOnMount: true`)

#### Dependency Management

The `fetchData` function is included in the refetch dependencies. To prevent infinite loops:

```tsx
// ✅ Good: Define fetchData outside component or wrap in useCallback
const fetchData = useCallback(async (params) => {
  const res = await fetch('/api/users', {
    method: 'POST',
    body: JSON.stringify(params),
  });
  return res.json();
}, []); // Empty deps - function is stable

const table = useDataTable({ fetchData });
```

```tsx
// ❌ Bad: Inline function (recreated on every render)
const table = useDataTable({
  fetchData: async (params) => { /* ... */ },
});
```

#### Loading States

The `isLoading` state is `true` during the initial fetch and any subsequent refetches. The `DataTable` component will display a skeleton loader when `isLoading` is `true`.

#### Error Recovery

Errors are caught and stored in the `error` field. The hook does not automatically retry on error - use the `refetch()` function to implement retry logic.

## Performance Considerations

### useIsMobile

- Uses `matchMedia` API for efficient resize detection (no resize event throttling needed)
- Only re-renders when crossing the breakpoint, not on every pixel change
- Minimal memory footprint (single event listener)

### useDataTable

- Automatically debounces filter updates (implement your own debouncing for text inputs)
- Uses `useCallback` internally for stable function references
- Only refetches when state actually changes (shallow equality check)

### Best Practices

```tsx
// ✅ Debounce search input for better UX
const [searchInput, setSearchInput] = useState('');
const debouncedSearch = useDebouncedValue(searchInput, 300);

useEffect(() => {
  table.setFilters(prev => ({ ...prev, search: debouncedSearch }));
}, [debouncedSearch]);

// ✅ Memoize fetchData to prevent unnecessary refetches
const fetchData = useCallback(async (params) => {
  // fetch logic
}, [/* dependencies */]);

// ✅ Reset to first page when filters change
const handleFilterChange = (newFilters) => {
  table.setFilters(newFilters);
  table.setPagination({ ...table.pagination, pageIndex: 0 });
};
```
